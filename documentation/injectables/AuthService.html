import { Injectable } from '@nestjs/common';
import { Request } from 'express';

@Injectable()
export class AuthService {
  googleLogin(req: Request): { message: string; user: any } {
    if (!req.user)
      return {
        message: 'no user from Google',
        user: {},
      };
    return {
      message: 'User information from googlee',
      user: req.user,
    };
  }
